<h2>The XMLHttpRequest Object</h2>
<button type="button" onclick="loadDoc()">Konuyu değiştir</button> <!-- buton ekliyoruz, tek tıklama ile loadDoc fonksiyonu cagrilir -->
</div>

<script>
function loadDoc() {
  const xhttp = new XMLHttpRequest();
  xhttp.onload = function() {
    document.getElementById("demo").innerHTML = 
    this.responseText;
  }
  xhttp.open("GET", "ajax_info.txt"); <!-- ajax_info.txt dosyasından metin alinir -->
  xhttp.send();
}
</script>
